file(GLOB_RECURSE modules ${PROJECT_SOURCE_DIR}/Buildscripts/CMake/Modules/*.cmake)

    include("${module}")
endforeach()

    Core
    Entities)

    # Savanna Core
    ${CMAKE_SOURCE_DIR}/Core

target_include_directories(
    ${app_name}
    PUBLIC
    ${CMAKE_CURRENT_LIST_DIR}
    # Core Libraries
    ${core_include_dirs})

# Testing Modules
add_subdirectory(Entities)
